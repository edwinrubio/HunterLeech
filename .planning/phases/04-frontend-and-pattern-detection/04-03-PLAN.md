```typescript
      {flags.map((flag) => (
        <FlagBadge key={flag} flag={flag} showTooltip />
      ))}
    </div>
  );
}
```

```typescript
interface ProfileHeaderProps {
  profile: ContractorProfile;
}

const TYPE_LABELS: Record<string, string> = {
  Empresa: "Empresa",
  Persona: "Persona Natural",
  EntidadPublica: "Entidad Publica",
};

const TYPE_COLORS: Record<string, string> = {
  Empresa: "bg-blue-100 text-blue-800",
  Persona: "bg-green-100 text-green-800",
  EntidadPublica: "bg-orange-100 text-orange-800",
};

export function ProfileHeader({ profile }: ProfileHeaderProps) {
  const identifier = profile.nit
    ? `NIT: ${profile.nit}`
    : profile.cedula
    ? `C.C.: ${profile.cedula}`
    : null;

  const fechaFmt = profile.ultima_actualizacion
    ? new Date(profile.ultima_actualizacion).toLocaleDateString("es-CO", {
        day: "2-digit",
        month: "short",
        year: "numeric",
      })
    : null;

  return (
    <div className="bg-white border-b border-slate-200 px-6 py-6">
      <div className="max-w-7xl mx-auto">
        <div className="flex items-start justify-between gap-4 mb-3">
          <div>
            <h1 className="text-2xl font-bold text-slate-900">{profile.nombre}</h1>
            {identifier && (
              <p className="text-sm font-mono text-slate-500 mt-0.5">{identifier}</p>
            )}
          </div>
          <div className="flex items-center gap-3 shrink-0">
            <span
              className={`px-2 py-1 rounded text-xs font-medium ${TYPE_COLORS[profile.type] ?? "bg-slate-100 text-slate-700"}`}
            >
              {TYPE_LABELS[profile.type] ?? profile.type}
            </span>
            <Link
              to={`/grafo/${encodeURIComponent(profile.id)}`}
              className="text-xs text-amber-600 hover:text-amber-800 underline"
            >
              Ver en grafo →
            </Link>
          </div>
        </div>

        <FlagSummaryBar flags={profile.flags} />

        {fechaFmt && (
          <p className="text-xs text-slate-400 mt-1">
            Datos actualizados: {fechaFmt} · Fuente:{" "}
            <span className="font-mono">{profile.fuente}</span>
          </p>
        )}
      </div>
    </div>
  );
}
```

```typescript
import { useEffect } from "react";
import { useLoadGraph } from "@react-sigma/core";
import { useLayoutForceAtlas2 } from "@react-sigma/layout-forceatlas2";
import { MultiDirectedGraph } from "graphology";
import type { GraphResponse, GraphNodeType } from "@/types/graph";

// Node type → Sigma attribute mapping
const NODE_COLORS: Record<GraphNodeType, string> = {
  Empresa: "#3B82F6",       // blue-500
  Persona: "#10B981",       // green-500
  EntidadPublica: "#F59E0B", // amber-500
  Contrato: "#6B7280",      // gray-500
  Sancion: "#EF4444",       // red-500
};

const NODE_SIZES: Record<GraphNodeType, number> = {
  Empresa: 12,
  Persona: 10,
  EntidadPublica: 14,
  Contrato: 6,
  Sancion: 8,
};

interface GraphLoaderProps {
  data: GraphResponse;
}

export function GraphLoader({ data }: GraphLoaderProps) {
  const loadGraph = useLoadGraph();
  const { assign: assignLayout } = useLayoutForceAtlas2({
    iterations: 150,
    settings: {
      gravity: 1,
      scalingRatio: 2,
      strongGravityMode: false,
    },
  });

  useEffect(() => {
    const graph = new MultiDirectedGraph();

    // Add nodes
    data.nodes.forEach((node) => {
      graph.addNode(node.id, {
        label: node.label,
        color: NODE_COLORS[node.type] ?? "#94A3B8",
        size: NODE_SIZES[node.type] ?? 8,
        nodeType: node.type,
        flags: node.flags ?? [],
        // Random initial positions — ForceAtlas2 will arrange them
        x: Math.random() * 100,
        y: Math.random() * 100,
      });
    });

    // Add edges
    data.edges.forEach((edge) => {
      // graphology may throw if source/target don't exist — guard
      if (graph.hasNode(edge.source) && graph.hasNode(edge.target)) {
        graph.addEdgeWithKey(edge.id, edge.source, edge.target, {
          label: edge.type,
          color: "#CBD5E1", // slate-300
          size: 1,
        });
      }
    });

    loadGraph(graph);

    // Run ForceAtlas2 layout after graph is loaded
    // Small delay ensures Sigma has mounted the graph before layout runs
    const timer = setTimeout(() => assignLayout(), 100);
    return () => clearTimeout(timer);
  }, [data, loadGraph, assignLayout]);

  return null;
}
```

```typescript
import { useState } from "react";
import { SigmaContainer, useRegisterEvents, useSigma } from "@react-sigma/core";
import { GraphLoader } from "./GraphLoader";
import { NodeSidebar } from "./NodeSidebar";
import { GraphControls } from "./GraphControls";
import type { GraphResponse, GraphNode } from "@/types/graph";

interface EventHandlerProps {
  graphData: GraphResponse;
  onNodeSelect: (node: GraphNode | null) => void;
}

function EventHandler({ graphData, onNodeSelect }: EventHandlerProps) {
  const registerEvents = useRegisterEvents();
  const sigma = useSigma();

  registerEvents({
    clickNode: ({ node: nodeId }) => {
      const nodeData = graphData.nodes.find((n) => n.id === nodeId);
      if (nodeData) {
        onNodeSelect(nodeData);
        // Highlight selected node
        sigma.getGraph().setNodeAttribute(nodeId, "highlighted", true);
      }
    },
    clickStage: () => {
      onNodeSelect(null);
    },
  });

  return null;
}

interface GraphExplorerProps {
  data: GraphResponse;
  depth: 1 | 2;
  onDepthChange: (depth: 1 | 2) => void;
}

export function GraphExplorer({ data, depth, onDepthChange }: GraphExplorerProps) {
  const [selectedNode, setSelectedNode] = useState<GraphNode | null>(null);

  return (
    <div className="relative w-full h-full">
      <SigmaContainer
        style={{ width: "100%", height: "100%" }}
        settings={{
          renderEdgeLabels: false,
          defaultEdgeColor: "#CBD5E1",
          labelThreshold: 6,
          minCameraRatio: 0.05,
          maxCameraRatio: 20,
          allowInvalidContainer: false,
        }}
      >
        <GraphLoader data={data} />
        <EventHandler graphData={data} onNodeSelect={setSelectedNode} />
      </SigmaContainer>

      <NodeSidebar
        node={selectedNode}
        onClose={() => setSelectedNode(null)}
      />

      <GraphControls
        depth={depth}
        onDepthChange={onDepthChange}
        nodeCount={data.nodes.length}
        edgeCount={data.edges.length}
      />
    </div>
  );
```
