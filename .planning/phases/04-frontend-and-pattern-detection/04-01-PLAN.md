```typescript
const BASE = "/api/v1";

export class ApiError extends Error {
  constructor(
    public status: number,
    message: string,
  ) {
    super(message);
    this.name = "ApiError";
  }
}

export async function apiFetch<T>(
  path: string,
  params?: Record<string, string>,
): Promise<T> {
  const url = new URL(BASE + path, window.location.origin);
  if (params) {
    Object.entries(params).forEach(([k, v]) => url.searchParams.set(k, v));
  }
  const res = await fetch(url.toString());
  if (!res.ok) {
    const text = await res.text();
    throw new ApiError(res.status, `Error ${res.status}: ${text}`);
  }
  return res.json() as Promise<T>;
}
```

```typescript
import { useQuery } from "@tanstack/react-query";
import { getMeta } from "@/api/meta";

export function FreshnessIndicator() {
  const { data } = useQuery({
    queryKey: ["meta"],
    queryFn: getMeta,
    staleTime: 60 * 60 * 1000, // Meta only changes after ETL runs — 1h cache
  });

  if (!data) return null;

  const sources = Object.entries(data.sources);
  const oldest = sources.reduce<string | null>((acc, [, src]) => {
    if (!acc) return src.last_ingested;
    return src.last_ingested < acc ? src.last_ingested : acc;
  }, null);

  if (!oldest) return null;

  const date = new Date(oldest).toLocaleDateString("es-CO", {
    day: "2-digit",
    month: "short",
    year: "numeric",
  });

  return (
    <span className="text-xs text-slate-400">
      Datos al <time dateTime={oldest}>{date}</time>
    </span>
  );
}
```

```typescript
import { useState, useEffect } from "react";

interface SearchBarProps {
  onSearch: (query: string) => void;
  placeholder?: string;
  autoFocus?: boolean;
}

function useDebounce<T>(value: T, delay: number): T {
  const [debounced, setDebounced] = useState(value);
  useEffect(() => {
    const timer = setTimeout(() => setDebounced(value), delay);
    return () => clearTimeout(timer);
  }, [value, delay]);
  return debounced;
}

export function SearchBar({
  onSearch,
  placeholder = "Buscar por NIT, cedula o nombre...",
  autoFocus = false,
}: SearchBarProps) {
  const [value, setValue] = useState("");
  const debounced = useDebounce(value, 300);

  useEffect(() => {
    onSearch(debounced);
  }, [debounced, onSearch]);

  return (
    <div className="relative w-full">
      <input
        type="search"
        value={value}
        onChange={(e) => setValue(e.target.value)}
        placeholder={placeholder}
        autoFocus={autoFocus}
        className="w-full px-4 py-3 pr-10 text-base border border-slate-300 rounded-lg shadow-sm
                   bg-white text-slate-900 placeholder:text-slate-400
                   focus:outline-none focus:ring-2 focus:ring-amber-500 focus:border-amber-500"
      />
      {value && (
        <button
          onClick={() => setValue("")}
          className="absolute right-3 top-1/2 -translate-y-1/2 text-slate-400 hover:text-slate-600"
          aria-label="Limpiar busqueda"
        >
          ✕
        </button>
      )}
    </div>
  );
}
```
